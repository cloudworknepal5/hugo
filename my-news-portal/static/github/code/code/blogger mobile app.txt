<script type='text/javascript'>
//<![CDATA[
/**
 * Birgunj Blog PWA - All-in-One Script
 * Features: Auto-Manifest, Service Worker, Auto-Install Prompt, iOS Support
 */
const BirgunjPWA = {
    deferredPrompt: null,
    config: {
        name: "Birgunj Blog",
        shortName: "Birgunj",
        icon: "https://blogger.googleusercontent.com/img/a/AVvXsEiBnC4hcDiMC18QrERnMrE8HTsMkzJDQqBmgeGvMpw_MA8NcKTPX3jUdY-byqu7K7iXUR9uByo0VBeiYdx5UXPJQHoslvzt6z-EprS-I-bg-L-w9hC_n2AUfIXuq5Nr5R1jZF5txT9r3_g5zq6FE1O8KcpaTVzrrhTWEIFv2PsjwJTSuLyHWRcjtzKRLnI",
        themeColor: "#2196f3"
    },

    // १. म्यानिफेस्टो र मेटा ट्यागहरू इन्जेक्सन गर्ने (Multi-function)
    initSetup: function() {
        const manifest = {
            "name": this.config.name,
            "short_name": this.config.shortName,
            "start_url": window.location.origin,
            "display": "standalone",
            "background_color": "#ffffff",
            "theme_color": this.config.themeColor,
            "icons": [{
                "src": this.config.icon,
                "sizes": "512x512",
                "type": "image/png",
                "purpose": "any maskable"
            }]
        };
        
        // Manifest Link
        const blob = new Blob([JSON.stringify(manifest)], {type: 'application/json'});
        const link = document.createElement('link');
        link.rel = 'manifest';
        link.href = URL.createObjectURL(blob);
        document.head.appendChild(link);

        // Apple Touch Icon
        const appleIcon = document.createElement('link');
        appleIcon.rel = 'apple-touch-icon';
        appleIcon.href = this.config.icon;
        document.head.appendChild(appleIcon);
        
        // Theme Color Meta
        const metaTheme = document.createElement('meta');
        metaTheme.name = 'theme-color';
        metaTheme.content = this.config.themeColor;
        document.head.appendChild(metaTheme);
    },

    // २. सर्भिस वर्कर दर्ता
    registerWorker: function() {
        if ('serviceWorker' in navigator) {
            const sw = 'data:application/javascript;base64,c2VsZi5hZGRFdmVudExpc3RlbmVyKCdmZXRjaCcsIGZ1bmN0aW9uKGV2ZW50KSB7IH0pOw==';
            navigator.serviceWorker.register(sw).then(() => console.log("PWA Registered"));
        }
    },

    // ३. इन्स्टल ब्यानर बनाउने र देखाउने
    createBanner: function() {
        const html = `
            <div id="pwa-banner">
                <img src="${this.config.icon}" alt="App Icon"/>
                <div class="text">
                    <div class="title">${this.config.name}</div>
                    <div class="desc">Home Screen मा थप्नुहोस्</div>
                </div>
                <button id="pwa-btn-install">Install</button>
                <button id="pwa-close">&times;</button>
            </div>`;
        document.body.insertAdjacentHTML('beforeend', html);

        const banner = document.getElementById('pwa-banner');
        
        // इन्स्टल इभेन्ट सुन्ने
        window.addEventListener('beforeinstallprompt', (e) => {
            e.preventDefault();
            this.deferredPrompt = e;
            // वेबसाइट खुलेको ३ सेकेन्डपछि ब्यानर देखाउने (Auto-Prompt)
            if (!localStorage.getItem('pwa_hide')) {
                setTimeout(() => banner.style.display = 'flex', 3000);
            }
        });

        document.getElementById('pwa-btn-install').addEventListener('click', () => {
            if (this.deferredPrompt) {
                this.deferredPrompt.prompt();
                this.deferredPrompt.userChoice.then(() => {
                    banner.style.display = 'none';
                    this.deferredPrompt = null;
                });
            }
        });

        document.getElementById('pwa-close').addEventListener('click', () => {
            banner.style.display = 'none';
            localStorage.setItem('pwa_hide', 'true'); // प्रयोगकर्ताले बन्द गरेपछि फेरि नदेखाउने
        });
    }
};

// सबै कामहरू सुरु गर्ने
window.addEventListener('DOMContentLoaded', () => {
    BirgunjPWA.initSetup();
    BirgunjPWA.registerWorker();
    BirgunjPWA.createBanner();
});
//]]>
</script>
