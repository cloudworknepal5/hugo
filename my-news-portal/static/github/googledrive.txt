<script>
const SCRIPT_BASE_URL = "https://script.google.com/macros/s/AKfycbzB6SCAGH8aOweGG6twgctfcyBg48eU-TCgNolqNOoGGp6DiTuBXJ6MjMiLwIVJ8o0U/exec";
const MY_FOLDER_ID = "133kF0VZL3_HiXEteCno3WOkqDqNEZaKW"; 
const PAGE_TITLE = "फेमस नेपाल सन्देश साप्ताहिक पढ्नुहोस्";

document.getElementById('gallery-title').innerText = PAGE_TITLE;

let allFiles = []; 
let displayedCount = 0; 
const BATCH_SIZE = 9; 

async function fetchAndInit() {
    const url = `${SCRIPT_BASE_URL}?id=${MY_FOLDER_ID}&t=${new Date().getTime()}`;
    try {
        const response = await fetch(url);
        const data = await response.json();
        if (data.error) { document.getElementById('status-msg').innerHTML = "त्रुटि: " + data.error; return; }
        allFiles = data.sort((a, b) => new Date(b.date) - new Date(a.date));
        document.getElementById('status-msg').style.display = 'none';
        renderBatch(); 
    } catch (err) {
        document.getElementById('status-msg').innerHTML = "डाटा लोड गर्न सकिएन।";
    }
}

function createCardHTML(file) {
    const thumb = `https://drive.google.com/thumbnail?id=${file.id}&sz=w1000`;
    // गुगल ड्राइभ डायरेक्ट डाउनलोड लिंक
    const downloadLink = `https://drive.google.com/uc?export=download&id=${file.id}`;
    
    return `
        <div class="drive-card-wrapper" style="position:relative">
            <a href="${downloadLink}" class="download-btn" title="Download ${file.name}" download>
                <svg viewBox="0 0 24 24"><path d="M5 20h14v-2H5v2zM19 9h-4V3H9v6H5l7 7 7-7z"/></svg>
            </a>
            <a href="${file.url}" target="_blank" class="drive-card">
                <div class="thumb-wrapper">
                    <img src="${thumb}" alt="${file.name}" loading="lazy">
                    <div class="file-name-overlay">${file.name}</div>
                </div>
            </a>
        </div>
    `;
}

function renderBatch() {
    const mainContainer = document.getElementById('main-gallery-container');
    const nextBatch = allFiles.slice(displayedCount, displayedCount + BATCH_SIZE);
    if (nextBatch.length > 0) {
        const groupDiv = document.createElement('div');
        groupDiv.className = 'grid-group';
        let batchHTML = '';
        nextBatch.forEach(file => { batchHTML += createCardHTML(file); });
        groupDiv.innerHTML = batchHTML;
        mainContainer.appendChild(groupDiv);
    }
    displayedCount += nextBatch.length;
    updateButtonVisibility();
}

function updateButtonVisibility() {
    const btn = document.getElementById('load-more-btn');
    btn.style.display = (displayedCount < allFiles.length) ? 'inline-block' : 'none';
}

function handleLoadMore() { renderBatch(); }
fetchAndInit();
</script>
